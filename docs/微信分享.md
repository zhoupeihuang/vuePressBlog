	/**
	 * 生成随机字符
	 * @param len 指定位数
	 * @returns {string}
	 */
	const randomStr = (len) =&gt; {
		/**
		 * 去掉了容易混淆的字符oOLl,9gq,Vv,Uu,I1
		 * @type {*|number}
		 */
		len = len || 32;
		let chars = 'ABCDEFGHJKMNPQRSTWXYZabcdefhijkmnprstwxyz2345678';
		let maxPos = chars.length;
		let str = '';
		for (let i = 0; i &lt; len; i++) {
			str += chars.charAt(Math.floor(Math.random() * maxPos));
		}
		return str;
	}
